Odin Usage:

    build with cmd "odin build <project root dir>"
    run with cmd "./<executable_name>"
        you can also run with cmd "odin run <project root dir>", which builds then runs, then deletes executable
    
    source:
        https://odin-lang.org/docs/overview/
        

Logging Util:

    Build:

        # NOTE: only use the below commands (faster compiling) for development. Use the commands in the build scripts for releasing production executables with performance optimizations turned on (and slower compiling).

        build c code (from project root dir):
            linux:
                rm -rf src/c/build/
                mkdir src/c/build/
                gcc -c src/c/c_bindings.c -o src/c/build/c_bindings.o
                ar rcs src/c/build/libc_bindings.a src/c/build/c_bindings.o
            windows:
                # from "x64 native Tools Command Prompt for VS" windows app
                rd /s /q src\c\build
                mkdir src\c\build
                cl -TC -c src\c\c_bindings.c /Fosrc\c\build\
                lib -nologo src\c\build\c_bindings.obj -out:src\c\build\c_bindings.lib
            source: https://odin-lang.org/news/binding-to-c/#example

        build odin code:
            linux:
                rm -rf build; mkdir build # odin build cmd doesn't autocreate build dir
                odin build examples/readme_example.odin -file -out:build/readme_example
                odin build examples/full_example.odin -file -out:build/full_example
            windows:
                NOTE: run below cmds from "x64 native Tools Command Prompt for VS" windows app
                rd /s /q build
                mkdir build
                odin build examples/readme_example.odin -file -out:build/readme_example.exe
                odin build examples/full_example.odin -file -out:build/full_example.exe

            # NOTE: building all files at once would require making them an odin collection cause each test/example has its own main function

    Run:
        linux:
            ./build/full_example # run from project root dir
            cd log; ./../build/readme_example # cd into log dir b/c readme_example creates log.txt in the current working directory the executable is run from
        windows:
            build\full_example.exe
            cd log
            ..\build\readme_example.exe
